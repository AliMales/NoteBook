# 对象原型链指南

```javascript
// 保护data对象 
const obj = (function() {
		const data = {
			a: 1,
			b: 2
		}
		return {
			get(key) {
				return data[key]
			}
		}
	})()

// hack 拿到整个树上的对象
Object.defineProperty(Object.prototype, 'valueOf', {
		get() {
			return this
		}
})

const data = obj.get('valueOf')
data.c='xxx'
console.log(data)
```
